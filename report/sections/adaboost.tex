\begin{theorem}{(useful theorem)}
\label{thm:main}
\begin{linenomath*}
	Let $X\in\mathbb{R}^{n\times d}, Y\in\mathbb{R}^n$. Further define $\beta^* = \operatorname{argmin}_{\beta\in R^d}\|X\beta-Y\|^2$.
	Then $$\|Y\|^2=\|X\beta^*\|^2 + \|X\beta^* - Y\|^2.$$
\end{linenomath*}

\end{theorem}
